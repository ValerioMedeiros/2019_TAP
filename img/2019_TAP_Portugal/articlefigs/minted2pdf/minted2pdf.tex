\begin{minted}{ocaml}
    IMPLEMENTATION RussianMult_i
    REFINES RussianMult
    CONCRETE_VARIABLES xx,yy,total
    INVARIANT xx |$\in$| |$\mathbb{N}$| |$\wedge$| yy |$\in$| |$\mathbb{N}$| |$\wedge$| 
        total |$\in$| |$\mathbb{N}$|
    INITIALISATION 
        xx,yy,total := 0,0,0
    OPERATIONS
      RussMult(aa,bb) = 
        xx:=aa; yy:=bb; total:=0;
        WHILE xx > 0 DO
          IF xx mod 2 = 1 THEN 
            total := total + yy END;
          xx := xx / 2; yy := yy * 2
        INVARIANT xx |$\in$| |$\mathbb{N}$| |$\wedge$|
            total+xx*yy = aa * bb
        VARIANT xx
        END
      END
    END
\end{minted}
